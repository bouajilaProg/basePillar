### 7. feat(shared:logger) – Centralized Logging Service

This package provides a standardized logging utility to stream logs to your centralized monitoring web app.

**File:** `packages/logger/package.json`
```json
{
  "name": "@repo/logger",
  "version": "0.0.0",
  "main": "./index.ts",
  "types": "./index.ts",
  "dependencies": {
    "axios": "^1.6.0"
  }
}
```

**File:** `packages/logger/index.ts`
```typescript
type LogLevel = 'info' | 'warn' | 'error' | 'debug';

interface LogPayload {
  appName: string;
  message: string;
  type: LogLevel;
  timestamp: string;
  metadata?: Record<string, any>;
}

export const createLogger = (appName: string, color: string = '#3b82f6') => {
  const consoleUrl = process.env.CENTRALIZED_CONSOLE_URL;

  const sendToConsole = async (type: LogLevel, message: string, data?: any) => {
    // Local development output with branding
    const timestamp = new Date().toISOString();
    console.log(
      `%c[${appName}] %c[${type.toUpperCase()}] %c${message}`,
      `color: ${color}; font-weight: bold;`,
      `color: #666;`,
      `color: inherit;`,
      data || ''
    );

    // Remote persistence to the logging web app
    if (consoleUrl) {
      try {
        const payload: LogPayload = {
          appName,
          message,
          type,
          timestamp,
          metadata: data,
        };
        // Using fetch to avoid heavy dependencies in frontend/backend
        await fetch(`${consoleUrl}/logs`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload),
        });
      } catch (err) {
        console.error('Failed to ship logs to centralized console:', err);
      }
    }
  };

  return {
    info: (msg: string, data?: any) => sendToConsole('info', msg, data),
    warn: (msg: string, data?: any) => sendToConsole('warn', msg, data),
    error: (msg: string, data?: any) => sendToConsole('error', msg, data),
    debug: (msg: string, data?: any) => sendToConsole('debug', msg, data),
  };
};
```
